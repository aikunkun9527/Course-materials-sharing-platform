## 二、阿里云 SAE (Serverless应用引擎) 集成

### 2.1 SAE 简介

阿里云 Serverless 应用引擎（SAE）是面向应用的 Serverless PaaS 平台，支持 Spring Cloud、Dubbo、HSF 等主流微服务框架，支持 Web 应用和微服务应用的全生命周期管理。

**优势**：
- 无需服务器管理
- 自动弹性伸缩
- 支持多种代码部署方式（镜像、代码包、JAR/WAR）
- 开箱即用的监控和日志
- 支持多种语言和框架

### 2.2 应用配置

#### 2.2.1 基本信息

#### 4.4.2 OSS 工具类

```javascript
// src/utils/oss.js
const OSS = require('ali-oss');

const ossClient = new OSS({
  region: process.env.OSS_REGION,
  accessKeyId: process.env.OSS_ACCESS_KEY_ID,
  accessKeySecret: process.env.OSS_ACCESS_KEY_SECRET,
  bucket: process.env.OSS_BUCKET_NAME,
  endpoint: process.env.OSS_ENDPOINT
});

/**
 * 生成文件路径
 */
function generateFilePath(type, userId, fileName) {
  const date = new Date();
  const year = date.getFullYear();
  const month = String(date.getMonth() + 1).padStart(2, '0');

  const uuid = require('crypto').randomUUID();
  const ext = fileName.split('.').pop();

  switch (type) {
    case 'avatar':
      return `avatars/${year}/${month}/user_${userId}.${ext}`;
    case 'material':
      return `materials/${year}/${month}/${uuid}.${ext}`;
    case 'course-cover':
      return `course-covers/${year}/${month}/${uuid}.${ext}`;
    default:
      return `uploads/temp/${uuid}/${fileName}`;
  }
}

/**
 * 上传文件（服务端上传）
 */
async function uploadFile(type, userId, file) {
  const filePath = generateFilePath(type, userId, file.originalname);

  try {
    const result = await ossClient.put(filePath, file.buffer);
    return {
      url: result.url,
      name: result.name,
      size: file.size
    };
  } catch (error) {
    throw new Error(`OSS upload failed: ${error.message}`);
  }
}

/**
 * 生成临时访问URL (签名URL)
 */
async function getSignedUrl(objectName, expires = 3600) {
  try {
    const url = ossClient.signatureUrl(objectName, {
      expires,
      method: 'GET'
    });
    return url;
  } catch (error) {
    throw new Error(`Generate signed URL failed: ${error.message}`);
  }
}

/**
 * 删除文件
 */
async function deleteFile(objectName) {
  try {
    await ossClient.delete(objectName);
    return true;
  } catch (error) {
    throw new Error(`Delete file failed: ${error.message}`);
  }
}

/**
 * 获取STS临时凭证（前端直传）
 */
async function getSTS_Token(userId) {
  const STS = require('ali-oss/lib/sts');

  const sts = new STS({
    accessKeyId: process.env.OSS_ACCESS_KEY_ID,
    accessKeySecret: process.env.OSS_ACCESS_KEY_SECRET
  });

  try {
    const result = await sts.assumeRole(
      process.env.OSS_ROLE_ARN,
      `{
        "Version": "1",
        "Statement": [
          {
            "Action": [
              "oss:PutObject"
            ],
            "Resource": [
              "acs:oss:*:*:course-platform-files/*"
            ],
            "Effect": "Allow"
          }
        ]
      }`,
      3000, // 过期时间（秒）
      'session-name'
    );

    return {
      region: process.env.OSS_REGION,
      accessKeyId: result.credentials.accessKeyId,
      accessKeySecret: result.credentials.accessKeySecret,
      stsToken: result.credentials.securityToken,
      bucket: process.env.OSS_BUCKET_NAME
    };
  } catch (error) {
    throw new Error(`Get STS token failed: ${error.message}`);
  }
}

module.exports = {
  uploadFile,
  getSignedUrl,
  deleteFile,
  getSTS_Token,
  generateFilePath
};
```

### 4.5 文件上传方案

#### 4.5.1 方案一：服务端上传（简单但占用带宽）

```javascript
// src/controllers/material.controller.js
const multer = require('multer');
const { uploadFile } = require('../utils/oss');
const Material = require('../models/material.model');

const upload = multer({ storage: multer.memoryStorage() });

exports.uploadMaterial = async (req, res) => {
  try {
    const { courseId, title, description } = req.body;
    const file = req.file;

    // 验证文件类型
    const allowedTypes = ['application/pdf', 'application/msword', 'video/mp4'];
    if (!allowedTypes.includes(file.mimetype)) {
      return res.status(400).json({ message: '不支持的文件类型' });
    }

    // 上传到 OSS
    const ossResult = await uploadFile('material', req.user.id, file);

    // 保存到数据库
    const material = await Material.create({
      course_id: courseId,
      uploader_id: req.user.id,
      title,
      description,
      file_url: ossResult.url,
      file_name: file.originalname,
      file_size: file.size,
      file_type: file.mimetype,
      file_extension: file.originalname.split('.').pop()
    });

    res.status(201).json({ success: true, data: material });
  } catch (error) {
    res.status(500).json({ message: error.message });
  }
};

// 路由
router.post('/materials',
  authenticate,
  upload.single('file'),
  uploadMaterial
);
```

```tsx
// components/common/FileUpload.tsx
import OSS from 'ali-oss';

const FileUpload = ({ courseId, onSuccess }) => {
  const handleUpload = async (file: File) => {
    try {
      // 1. 获取上传签名

        fileName: file.name,
        fileSize: file.size,

        accessKeySecret: data.accessKeySecret,
        stsToken: data.stsToken,
        bucket: data.bucket
      });

      // 3. 上传文件

        fileType: file.type
      });

      message.success('上传成功');
      onSuccess();
    } catch (error) {
      message.error('上传失败');
    }
  };
```

```javascript
exports.downloadMaterial = async (req, res) => {
```
